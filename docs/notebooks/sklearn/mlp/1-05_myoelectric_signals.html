parameters = [
    {
        "hidden_layer_sizes": [(h,) for h in range(5)],
        "learning_rate_init": [
            0.01,
            0.05,
            0.10,

            0.25,
            0.30,
        ],
        "momentum": [
            0.60,
            0.70,
            0.80,
            0.90,
        ],
    },
]


m = GridSearchCV(
    MLPClassifier(
        activation="logistic",
        learning_rate="adaptive",
        max_iter=1000,
        verbose=0,
    ),
    parameters,
    cv=10,
)
