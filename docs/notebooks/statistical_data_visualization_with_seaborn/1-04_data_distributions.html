[97]:

#
# Modificación de las propiedades
#
p = sns.boxplot(
    x=penguins.flipper_length_mm,
    color='lightblue',
)

for e in p.artists:
    r, g, b, a = e.get_facecolor()
    p.set_facecolor((r, g, b, 0.0))

for i, artist in enumerate(p.artists):
    artist.set_edgecolor("blue")

for j in range(0, 6):
    line = p.lines[j]
    line.set_color("red")

#
# Uso de JoinGrid para espeficiar el tipo de gráficas
# que se usaran en cada parte
#
g = sns.JointGrid(data=penguins, x="bill_length_mm", y="bill_depth_mm")
g.plot_joint(sns.histplot)
g.plot_marginals(sns.boxplot)

plt.show()

[42]:

#
# Uso de PairGrid() para especificar el tipo de
# gráfico en cada parte
#
g = sns.PairGrid(penguins)
g.map_upper(sns.histplot)
g.map_lower(sns.kdeplot, fill=True)
g.map_diag(sns.histplot, kde=True)
plt.show()
