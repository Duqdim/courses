#
g = sns.catplot(
    x="day",
    y="total_bill",
    kind="violin",
    inner=None,
    data=tips,
)

sns.swarmplot(
    x="day",
    y="total_bill",
    color="k",
    size=3,
    data=tips,
    ax=g.ax,
)

plt.show()
