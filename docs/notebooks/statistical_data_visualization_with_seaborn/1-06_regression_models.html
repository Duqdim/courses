Funciones para dibujar modelos de regresión lineal¶

[28]:

#
# Regresión univariada con intervalos de
# confianza del 95% usando regplot().
#
sns.regplot(
    x="total_bill",
    y="tip",
    data=tips,
    marker='+',
)

plt.show()

[4]:

#
# Gráfica equivalente usando lmplot()
#
sns.lmplot(
    x="total_bill",
    y="tip",
    data=tips,
    x_jitter=0.05,
)

plt.show()

[5]:

#
# Visualización de los datos como un intervalo
#
sns.lmplot(
    x="size",
    y="tip",
    data=tips,
    x_estimator=np.mean,
)

plt.show()

#
# Regresión logística.
#
tips["big_tip"] = (tips.tip / tips.total_bill) > .15

sns.lmplot(
    x="total_bill",
    y="big_tip",
    data=tips,
    logistic=True,
    y_jitter=0.03,
)

plt.show()
