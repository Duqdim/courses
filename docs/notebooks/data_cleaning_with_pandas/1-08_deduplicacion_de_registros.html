#
# Comparación de registros
#
compare_cl = recordlinkage.Compare()

compare_cl.string(
    "first_name",
    "first_name",
    method="levenshtein",
    threshold=0.50,
    label="first_name",
)

compare_cl.string(
    "last_name",
    "last_name",
    method="levenshtein",
    threshold=0.50,
    label="last_name",
)

features = compare_cl.compute(candidate_links, df)
